## LC 61. Rotate List

```python
# Definition for singly-linked list.
# class ListNode:
#     def __init__(self, x):
#         self.val = x
#         self.next = None

class Solution:
    def rotateRight(self, head: ListNode, k: int) -> ListNode:
        #Solution
        #解题思路：循环右移一条链表，比如k=2，（1，2，3，4，5）循环右移两位变为（4，5，1，2，3）。
        #由于k值有可能比链表长度大很多，所以先要用一个count变量求出链表的长度。而k%count就是循环右移的步数。
        #1->2->3->4->5,k=2
        if k == 0 or head == None:
            return head
        dummy = ListNode(0)
        dummy.next = head
        p = dummy
        count = 0
        while p.next:
            p = p.next
            count += 1
        #此时p指向最后一个结点，这里先和head接上，变成一个环
        #1->2->3->4->5->连到1形成环
        p.next = dummy.next
        step = count - (k % count)
        #找到新的头结点的前一个结点，摘链
        for _ in range(step):
            p = p.next
        #p此时等于3
        #新的head=4，4->5->1->2->3->None
        head = p.next
        p.next = None
        return head
```
