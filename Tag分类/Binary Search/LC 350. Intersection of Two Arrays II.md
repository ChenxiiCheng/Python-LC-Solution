class Solution:
    def intersect(self, nums1: List[int], nums2: List[int]) -> List[int]:
        #Solution 2 - ä¸ç”¨Counter
        counts = {}
        res = []
        for num in nums2:
            counts[num] = counts.get(num, 0) + 1
        for num in nums1:
            if num in counts and counts[num] > 0:
                res.append(num)
                counts[num] -= 1
        return res
