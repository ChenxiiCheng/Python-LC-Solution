#### Solution

```python
# Definition for a binary tree node.
# class TreeNode:
#     def __init__(self, x):
#         self.val = x
#         self.left = None
#         self.right = None

class Solution:
    def isSymmetric(self, root):
        """
        :type root: TreeNode
        :rtype: bool
        """
        #Solution
        if not root:
            return True
        stack = [(root.left, root.right)]
        while stack:
            n1, n2 = stack.pop()
            if not n1 and not n2:
                continue
            elif n1 and n2 and n1.val == n2.val:
                stack.append((n1.left, n2.right))
                stack.append((n1.right, n2.left))
            else:
                return False
        return True
```
